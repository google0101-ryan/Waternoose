project(DOOM3BFG)

set(CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/cmake)

			src/loader/xex.cpp

			src/cpu/ops.cpp)

include_directories(${CMAKE_SOURCE_DIR}/src)
add_executable(xbox360 ${SOURCES} ${AES_SOURCES})
